# Manual iteration on fields
n = 0
while True:
    f = r.iter_fields()
    if f is None:
        break
    elif n == 0:
        print(type(f))
    n += 1
print(n)

# Manual iterator on messages; returns epygram.formats.GRIB.GRIBmessage objects, i.e. data is not decoded
n = 0
while True:
    m = r.iter_messages()
    if m is None:
        break
    elif n == 0:
        print(type(m))
    n += 1
print(n)
